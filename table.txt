#include <iostream>
using namespace std;
void printTable(int j) {
    for (int i = 1; i <= 10; i++) {
        cout << j << " x " << i << " = " << j * i << endl;
    }
}
int main() {
    int num;
    cout << "Enter a number for table: ";
    cin >> num;
    printTable(num);
    return 0;
}
